async function loadArticles() { const grid = document.getElementById('mainGrid'); try { // قراءة الملف لي صاوب بايثون const response = await fetch('articles.json'); const articles = await response.json(); // مسح الكرت التجريبي grid.innerHTML = ''; articles.forEach((art, index) => { const card = document.createElement('div'); card.className = 'article-card'; card.innerHTML = `
${art.category || 'SEO'}

${art.title}

${art.snippet}

`; grid.appendChild(card); // تأثير التحميل (Skeleton) كيبقى ثانية واحدة عاد كيبان المحتوى setTimeout(() => { card.classList.add('content-ready'); }, 1000 + (index * 200)); // كل كرت كيبان مورا لآخر بشوية }); } catch (error) { console.error("Error loading articles:", error); } } window.onload = loadArticles;
